cmake_minimum_required(VERSION 3.16)
project(GeoBase)

# C++标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 解决中文编码警告
add_compile_options(/utf-8)

# 头文件路径
include_directories(
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_SOURCE_DIR}/thirdparty
)

# ======================
# 通配符 * 自动匹配所有文件
# 不用手动写，新增文件自动识别
# ======================
file(GLOB ALL_FILES
    # src下所有cpp
    ${PROJECT_SOURCE_DIR}/src/*.cpp
    # 头文件
    ${PROJECT_SOURCE_DIR}/include/GeoBase/*.h
    # glad源码
    ${PROJECT_SOURCE_DIR}/thirdparty/glad/*.c
)

# 生成程序
add_executable(GeoBase ${ALL_FILES})

# 链接GLFW
find_package(glfw3 CONFIG REQUIRED)
target_link_libraries(GeoBase PRIVATE glfw)

# 链接GTest（解决你刚才的链接报错！）
find_package(GTest CONFIG REQUIRED)
target_link_libraries(GeoBase PRIVATE GTest::gtest GTest::gtest_main)
